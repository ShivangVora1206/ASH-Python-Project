def w_next():
    global word_hist_index
    if word_hist_index < len(word_hist) - 1:
        word_hist_index += 1
    else: word_hist_index = 0
    return word_hist_index

def w_prev():
    global word_hist_index
    if word_hist_index > 0:
        word_hist_index -= 1
    else: word_hist_index = len(word_hist) - 1
    return word_hist_index
